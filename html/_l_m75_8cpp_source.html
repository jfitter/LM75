102 /**
103  * \brief Write the specified setpoint supplied in the specified units.
104  * \remarks Setpoint is written as a 2's complement value in the most significant
105  * 9 bits of a 16 bit field expressed in degrees C to a resolution of 0.125C.
106  * \param [in] val Setpoint temperature.
107  * \param [in] tunit Temperature units of the setpoint.
108  * \param [in] spt Setpoint type (overtemp or hysteresis)
109  */
110 void LM75::writeSetPoint(double val, tempUnit_t tunit, setPointType spt) {
111  uint16_t data;
112 
113  switch(tunit) {
114  case LM75_TK : val = convKtoC(val); break;
115  case LM75_TF : val = convFtoC(val);
116  }
117  data = (uint16_t)(val * 128.0/0.125 + 0.5);
118  write16(spt == SPT_OVERTEMP ? LM75_TOS : LM75_THYST, data);
119 }

175  * \param [in] reg 8b Register to read from.
176  * \return Value read from register.
177  */
178 uint8_t LM75::read8(uint8_t reg) {
179 
180  // send the device address then the register pointer byte
181  Wire.beginTransmission(_addr);
182  Wire.write(reg);
183  Wire.endTransmission(false);
184 
185  // resend device address then get the returned byte
186  Wire.requestFrom(_addr, (uint8_t)1);
187 
188  // data is returned as 1 byte
189  return Wire.read();
190 }

192 /**
193  * \brief Return a 16 bit register value from the device.
194  * \param [in] reg 16b Register to read from.
195  * \return Value read from register.
196  */
197 uint16_t LM75::read16(uint8_t reg) {
198  uint16_t val;
199 
200  // send the device address then the register pointer byte
201  Wire.beginTransmission(_addr);
202  Wire.write(reg);
203  Wire.endTransmission(false);
204 
205  // resend device address then get the 2 returned bytes
206  Wire.requestFrom(_addr, (uint8_t)2);
207 
208  // data is returned as 2 bytes big endian
209  val = Wire.read() << 8;
210  val |= Wire.read();
211 
212  return val;
213 }
